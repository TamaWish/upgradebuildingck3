# Confirm or cancel an automated upgrade after the dry-run cost preview.
abu_event.0002 = {
	type = character_event
	title = abu_upgrade_confirm_title
	desc = abu_upgrade_confirm_desc
	theme = stewardship
	left_portrait = root

	option = {
		name = abu_upgrade_confirm_proceed
		custom_tooltip = abu_upgrade_confirm_proceed_tt
		hidden_effect = {
			abu_upgrade_player_domain_buildings_effect = yes
			remove_variable = abu_upgrade_mode
			remove_variable = abu_upgrade_scope
			remove_variable = abu_upgrade_depth
			abu_clear_upgrade_preview_variables_effect = yes
		}
	}

	option = {
		name = abu_upgrade_confirm_cancel
		custom_tooltip = abu_upgrade_confirm_cancel_tt
		hidden_effect = {
			remove_variable = abu_upgrade_mode
			remove_variable = abu_upgrade_scope
			remove_variable = abu_upgrade_depth
			abu_clear_upgrade_preview_variables_effect = yes
		}
	}
}
