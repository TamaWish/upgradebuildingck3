#####################################################################
# Automated Building Upgrade - scripted effects
# Upgrades existing buildings in the player's direct domain only.
#####################################################################

abu_upgrade_player_domain_buildings_effect = {
	set_variable = {
		name = abu_upgrades_performed
		value = 0
	}
	set_variable = {
		name = abu_starting_gold
		value = gold
	}
	set_variable = {
		name = abu_gold_spent
		value = 0
	}

	while = {
		limit = {
			abu_has_affordable_upgrade_in_domain_trigger = yes
		}
		count = 48
		abu_upgrade_domain_one_pass_effect = yes
	}

	if = {
		limit = {
			var:abu_upgrades_performed > 0
		}
		set_variable = {
			name = abu_gold_spent
			value = {
				value = var:abu_starting_gold
				subtract = gold
			}
		}
		send_interface_message = {
			type = event_generic_good
			title = abu_upgrade_complete_title
			desc = abu_upgrade_complete_desc
			right_icon = root
		}
	}
	else = {
		send_interface_message = {
			type = event_generic_neutral
			title = abu_upgrade_none_title
			desc = abu_upgrade_none_desc
			right_icon = root
		}
	}

	remove_variable = abu_upgrades_performed
	remove_variable = abu_starting_gold
	remove_variable = abu_gold_spent
}

# One pass: raise every eligible building in the player's direct domain by one tier.
# Capital county is processed first so limited gold goes to the seat of power.
abu_upgrade_domain_one_pass_effect = {
	every_directly_owned_province = {
		limit = {
			county = root.capital_county
			abu_province_has_upgradable_building_trigger = yes
		}
		abu_upgrade_province_buildings_effect = yes
	}
	every_directly_owned_province = {
		limit = {
			NOT = { county = root.capital_county }
			abu_province_has_upgradable_building_trigger = yes
		}
		abu_upgrade_province_buildings_effect = yes
	}
}
