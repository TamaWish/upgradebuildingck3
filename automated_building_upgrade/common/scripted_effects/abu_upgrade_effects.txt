abu_spend_upgrade_gold_effect = {
	if = {
		limit = {
			has_variable = abu_dry_run
			var:abu_dry_run = 1
		}
		change_variable = {
			name = abu_simulated_gold
			subtract = $COST$
		}
	}
	else = {
		remove_short_term_gold = $COST$
	}
}

abu_begin_upgrade_with_preview_effect = {
	abu_estimate_upgrade_cost_effect = yes
	trigger_event = abu_event.0002
}

abu_estimate_upgrade_cost_effect = {
	every_directly_owned_province = {
		limit = {
			abu_province_in_upgrade_scope_trigger = yes
			NOT = { has_ongoing_construction = yes }
		}
		abu_snapshot_province_buildings_effect = yes
	}

	clear_variable_list = abu_upgraded_provinces
	set_variable = {
		name = abu_upgrades_performed
		value = 0
	}
	set_variable = {
		name = abu_main_upgrades
		value = 0
	}
	set_variable = {
		name = abu_slot_upgrades
		value = 0
	}
	set_variable = {
		name = abu_estimate_provinces
		value = 0
	}
	set_variable = {
		name = abu_dry_run
		value = 1
	}
	set_variable = {
		name = abu_simulated_gold
		value = gold
	}
	set_variable = {
		name = abu_starting_gold
		value = gold
	}

	abu_fully_upgrade_domain_effect = yes

	set_variable = {
		name = abu_estimate_cost
		value = {
			value = var:abu_starting_gold
			subtract = var:abu_simulated_gold
		}
	}
	set_variable = {
		name = abu_estimate_upgrades
		value = var:abu_upgrades_performed
	}

	every_directly_owned_province = {
		limit = {
			abu_province_in_upgrade_scope_trigger = yes
			has_variable = abu_snapshotted
		}
		abu_revert_province_from_snapshot_effect = yes
		abu_clear_province_snapshot_variables_effect = yes
	}

	remove_variable = abu_dry_run
	remove_variable = abu_simulated_gold
	remove_variable = abu_starting_gold
	remove_variable = abu_upgrades_performed
	remove_variable = abu_main_upgrades
	remove_variable = abu_slot_upgrades
	clear_variable_list = abu_upgraded_provinces
}

# Restore slot buildings after a dry-run. remove_building deletes the whole
# chain — use add_building to downgrade back to the snapshotted tier.
abu_revert_slot_building_to_snapshot_effect = {
	if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 0
			has_building_or_higher = $BUILDING$_01
		}
		remove_building = $BUILDING$_01
		remove_building = $BUILDING$_02
		remove_building = $BUILDING$_03
		remove_building = $BUILDING$_04
		remove_building = $BUILDING$_05
		remove_building = $BUILDING$_06
		remove_building = $BUILDING$_07
		remove_building = $BUILDING$_08
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 1
			has_building_or_higher = $BUILDING$_02
		}
		add_building = $BUILDING$_01
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 2
			has_building_or_higher = $BUILDING$_03
		}
		add_building = $BUILDING$_02
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 3
			has_building_or_higher = $BUILDING$_04
		}
		add_building = $BUILDING$_03
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 4
			has_building_or_higher = $BUILDING$_05
		}
		add_building = $BUILDING$_04
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 5
			has_building_or_higher = $BUILDING$_06
		}
		add_building = $BUILDING$_05
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 6
			has_building_or_higher = $BUILDING$_07
		}
		add_building = $BUILDING$_06
	}
	else_if = {
		limit = {
			has_variable = abu_snap_$BUILDING$
			var:abu_snap_$BUILDING$ = 7
			has_building_or_higher = $BUILDING$_08
		}
		add_building = $BUILDING$_07
	}
}

# Fully upgrade each directly owned province in scope. Capital counties go first in full-domain mode.
# Mode 0 = all, 1 = holdings only, 2 = economic slots, 3 = military slots.
# Scope 0/unset = domain, 1 = capital county, 2 = current duchy.
# Depth 0/unset = max, 1 = one tier, 2 = empty slots only, 3 = tier cap.
abu_fully_upgrade_domain_effect = {
	if = {
		limit = {
			abu_upgrade_scope_is_capital_county_trigger = yes
		}
		every_directly_owned_province = {
			limit = {
				abu_province_in_capital_county_trigger = yes
				NOT = { has_ongoing_construction = yes }
			}
			abu_fully_upgrade_province_effect = yes
		}
	}
	else_if = {
		limit = {
			abu_upgrade_scope_is_current_duchy_trigger = yes
		}
		every_directly_owned_province = {
			limit = {
				abu_province_in_current_duchy_trigger = yes
				NOT = { has_ongoing_construction = yes }
			}
			abu_fully_upgrade_province_effect = yes
		}
	}
	else = {
		every_directly_owned_province = {
			limit = {
				county = root.capital_county
				NOT = { has_ongoing_construction = yes }
			}
			abu_fully_upgrade_province_effect = yes
		}
		every_directly_owned_province = {
			limit = {
				NOT = { county = root.capital_county }
				NOT = { has_ongoing_construction = yes }
			}
			abu_fully_upgrade_province_effect = yes
		}
	}
}

abu_construct_one_free_slot_building_effect = {
	if = {
		limit = {
			root = { var:abu_upgrade_mode = 0 }
		}
		if = {
			limit = {
				has_variable = abu_prefer_military_slot
				var:abu_prefer_military_slot = 1
			}
			if = {
				limit = {
					abu_military_slot_building_can_construct_trigger = yes
				}
				abu_construct_best_military_slot_building_effect = yes
			}
			else_if = {
				limit = {
					abu_economic_slot_building_can_construct_trigger = yes
				}
				abu_construct_best_economic_slot_building_effect = yes
			}
			remove_variable = abu_prefer_military_slot
		}
		else = {
			if = {
				limit = {
					abu_economic_slot_building_can_construct_trigger = yes
				}
				abu_construct_best_economic_slot_building_effect = yes
			}
			else_if = {
				limit = {
					abu_military_slot_building_can_construct_trigger = yes
				}
				abu_construct_best_military_slot_building_effect = yes
			}
			set_variable = {
				name = abu_prefer_military_slot
				value = 1
			}
		}
	}
	else_if = {
		limit = {
			root = { var:abu_upgrade_mode = 2 }
		}
		abu_construct_best_economic_slot_building_effect = yes
	}
	else_if = {
		limit = {
			root = { var:abu_upgrade_mode = 3 }
		}
		abu_construct_best_military_slot_building_effect = yes
	}
}

abu_log_upgrade_province_effect = {
	save_scope_as = abu_upgrade_province
	root = {
		if = {
			limit = {
				NOT = {
					is_target_in_variable_list = {
						name = abu_upgraded_provinces
						target = scope:abu_upgrade_province
					}
				}
			}
			add_to_variable_list = {
				name = abu_upgraded_provinces
				target = scope:abu_upgrade_province
			}
			if = {
				limit = {
					has_variable = abu_dry_run
					var:abu_dry_run = 1
				}
				change_variable = {
					name = abu_estimate_provinces
					add = 1
				}
			}
			else = {
				change_variable = {
					name = abu_upgraded_province_count
					add = 1
				}
			}
		}
	}
}

	remove_variable = abu_upgrade_scope
	remove_variable = abu_upgrade_depth
	set_variable = { name = abu_upgrade_mode value = 0 }
	abu_upgrade_player_domain_buildings_effect = yes
	remove_variable = abu_upgrade_mode
}

abu_passive_auto_upgrade_innovation_pulse_effect = {
	abu_count_building_innovations_effect = yes
	if = {
		limit = {
			abu_should_passive_auto_upgrade_trigger = yes
		}
		abu_passive_auto_upgrade_execute_effect = yes
	}
	# Keep retrying after a new innovation if gold was too low; only sync when idle.
	if = {
		limit = {
			OR = {
				AND = {
					abu_passive_auto_upgrade_character_trigger = yes
					abu_culture_gained_building_innovation_trigger = yes
					abu_has_affordable_upgrade_in_domain_trigger = yes
				}
				NOT = { abu_culture_gained_building_innovation_trigger = yes }
			}
		}
		set_variable = {
			name = abu_tracked_innovation_count
			value = var:abu_building_innovation_count
		}
	}
	remove_variable = abu_building_innovation_count
}
