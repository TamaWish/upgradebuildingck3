abu_set_gold_reserve = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_spend_money.dds"
	}

	decision_group_type = abu_decisions
	ai_check_interval = 0

	is_shown = {
		is_ai = no
		is_playable_character = yes
		is_landed = yes
	}

	is_valid = {
		always = yes
	}

	effect = {
		custom_tooltip = abu_set_gold_reserve_tooltip
		hidden_effect = {
			trigger_event = abu_event.0001
		}
	}

	ai_will_do = { base = 0 }
}

abu_toggle_passive_auto_upgrade = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_spend_money.dds"
	}

	decision_group_type = abu_decisions
	ai_check_interval = 0

	is_shown = {
		is_ai = no
		is_playable_character = yes
		is_landed = yes
	}

	is_valid = {
		always = yes
	}

	effect = {
		custom_tooltip = abu_toggle_passive_auto_upgrade_tooltip
		if = {
			limit = {
				has_variable = abu_passive_auto_upgrade
				var:abu_passive_auto_upgrade = 1
			}
			remove_variable = abu_passive_auto_upgrade
			custom_tooltip = abu_passive_auto_upgrade_disabled_tt
		}
		else = {
			set_variable = {
				name = abu_passive_auto_upgrade
				value = 1
			}
			abu_update_tracked_innovation_count_effect = yes
			custom_tooltip = abu_passive_auto_upgrade_enabled_tt
		}
	}

	ai_will_do = { base = 0 }
}

abu_upgrade_to_tier_cap = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_spend_money.dds"
	}

	decision_group_type = abu_decisions
	ai_check_interval = 0

	is_shown = {
		is_ai = no
		is_playable_character = yes
		is_landed = yes
	}

	is_valid = {
		abu_has_affordable_tier_cap_upgrade_in_domain_trigger = yes
	}

	is_valid_showing_failures_only = {
		custom_tooltip = {
			text = abu_no_upgradable_buildings_tt
			abu_has_upgradable_building_in_domain_trigger = yes
		}
		custom_tooltip = {
			text = abu_insufficient_gold_tt
			abu_has_affordable_tier_cap_upgrade_in_domain_trigger = yes
		}
	}

	effect = {
		custom_tooltip = abu_upgrade_to_tier_cap_effect_tt
		hidden_effect = {
			remove_variable = abu_upgrade_scope
			if = {
				limit = {
					NOT = { has_variable = abu_tier_cap }
				}
				set_variable = {
					name = abu_tier_cap
					value = 4
				}
			}
			set_variable = { name = abu_upgrade_mode value = 0 }
			set_variable = { name = abu_upgrade_depth value = 3 }
			abu_begin_upgrade_with_preview_effect = yes
		}
	}

	ai_will_do = { base = 0 }
}
