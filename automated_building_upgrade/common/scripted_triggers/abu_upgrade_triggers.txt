# Player domain only - checks if a province has at least one building that can be upgraded.
abu_province_has_upgradable_building_trigger = {
	NOT = { has_ongoing_construction = yes }
	OR = {
		abu_main_holding_can_upgrade_trigger = yes
		abu_slot_building_can_upgrade_trigger = yes
	}
}

abu_main_holding_can_upgrade_trigger = {
	OR = {
		AND = {
			has_holding_type = castle_holding
			has_building = castle_01
			NOT = { has_building_or_higher = castle_02 }
			culture = { has_innovation = innovation_battlements }
		}
		AND = {
			has_holding_type = castle_holding
			has_building = castle_02
			NOT = { has_building_or_higher = castle_03 }
			culture = { has_innovation = innovation_hoardings }
		}
		AND = {
			has_holding_type = castle_holding
			has_building = castle_03
			NOT = { has_building_or_higher = castle_04 }
			culture = { has_innovation = innovation_machicolations }
		}
		AND = {
			has_holding_type = city_holding
			has_building = city_01
			NOT = { has_building_or_higher = city_02 }
			culture = { has_innovation = innovation_city_planning }
		}
		AND = {
			has_holding_type = city_holding
			has_building = city_02
			NOT = { has_building_or_higher = city_03 }
			culture = { has_innovation = innovation_manorialism }
		}
		AND = {
			has_holding_type = city_holding
			has_building = city_03
			NOT = { has_building_or_higher = city_04 }
			culture = { has_innovation = innovation_guilds }
		}
		AND = {
			has_holding_type = church_holding
			has_building = temple_01
			NOT = { has_building_or_higher = temple_02 }
			culture = { has_innovation = innovation_city_planning }
		}
		AND = {

			has_building = tribe_01
			NOT = { has_building_or_higher = tribe_02 }
		}
		AND = {
			has_holding_type = tribal_holding
			has_building = tribe_02

abu_can_afford_economic_slot_upgrade_trigger = {
	OR = {
		AND = {
			has_building = $BUILDING$_01
			NOT = { has_building_or_higher = $BUILDING$_02 }
			building_$BUILDING$_requirement_terrain = yes
			generic_economic_building_innovation_trigger = { BUILDING = $BUILDING$ }
			province_owner = { gold >= normal_building_tier_2_cost }
		}
		AND = {
			has_building = $BUILDING$_02
			NOT = { has_building_or_higher = $BUILDING$_03 }
			building_$BUILDING$_requirement_terrain = yes
