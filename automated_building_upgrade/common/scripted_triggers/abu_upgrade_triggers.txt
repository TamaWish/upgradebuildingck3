# Upgrade mode variable on character: 0 = all, 1 = holdings, 2 = economic, 3 = military
# Upgrade scope variable: unset/0 = full domain, 1 = capital county, 2 = current duchy
# Upgrade depth variable: unset/0 = max, 1 = one tier, 2 = empty slots only, 3 = tier cap
# Tier cap variable: abu_tier_cap (default 4 when depth = tier cap)
# Gold reserve variable on character: abu_gold_reserve (unset or 0 = spend until broke)
# Passive auto-upgrade: abu_passive_auto_upgrade (1 = enabled)
# Duchy capital buildings: abu_include_duchy_capital (1 = include existing duchy buildings in upgrades)
# Innovation tracking: abu_tracked_innovation_count, abu_building_innovation_count (pulse scratch)

abu_character_can_afford_cost_trigger = {
	OR = {
		AND = {
			has_variable = abu_dry_run
			var:abu_dry_run = 1
			OR = {
				AND = {
					OR = {
						NOT = { has_variable = abu_gold_reserve }
						var:abu_gold_reserve = 0
					}
					var:abu_simulated_gold >= $COST$
				}
				AND = {
					has_variable = abu_gold_reserve
					var:abu_gold_reserve > 0
					var:abu_simulated_gold >= {
						value = $COST$
						add = var:abu_gold_reserve
					}
				}
			}
		}
		AND = {
			OR = {
				NOT = { has_variable = abu_dry_run }
				NOT = { var:abu_dry_run = 1 }
			}
			OR = {
				AND = {
					OR = {
						NOT = { has_variable = abu_gold_reserve }
						var:abu_gold_reserve = 0
					}
					gold >= $COST$
				}
				AND = {
					has_variable = abu_gold_reserve
					var:abu_gold_reserve > 0
					gold >= {
						value = $COST$
						add = var:abu_gold_reserve
					}
				}
			}
		}
	}
}

# Player domain only - checks if a province has at least one building that can be upgraded or constructed.
abu_province_has_upgradable_building_trigger = {
	NOT = { has_ongoing_construction = yes }
	OR = {
		abu_main_holding_can_upgrade_trigger = yes
		abu_economic_slot_building_can_upgrade_trigger = yes
		abu_military_slot_building_can_upgrade_trigger = yes
		abu_economic_slot_building_can_construct_trigger = yes
		abu_military_slot_building_can_construct_trigger = yes
		abu_duchy_capital_building_can_upgrade_trigger = yes
	}
}
